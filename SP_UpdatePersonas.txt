CREATE PROCEDURE SP_UpdatePersona
(
	@Id int = 0,
	@ID_PACIENTE_TITULAR varchar(150) = '',
	@ID_VENDOR int = 0,
	@ID_PERSONAL_PACIENTE int = 0,
	@NUMERO_CARNET int = 0,
	@NOMBRE varchar(150) = '',
	@PRIMER_APELLIDO varchar(150) = '',
	@SEGUNDO_APELLIDO varchar(150) = '',
	@PAIS varchar(150) = '',
)
as
BEGIN
	Update Personas Set 
	ID_PACIENTE_TITULAR=@ID_PACIENTE_TITULAR,
	ID_VENDOR=@ID_VENDOR,
	ID_PERSONAL_PACIENTE=@ID_PERSONAL_PACIENTE,
	NUMERO_CARNET=@NUMERO_CARNET,
	NOMBRE=@NOMBRE,
	PRIMER_APELLIDO=@PRIMER_APELLIDO,
	SEGUNDO_APELLIDO=@SEGUNDO_APELLIDO,
	WHERE Id=@Id
END

go

CREATE PROCEDURE SP_UpdateProducto
(
	@Id int = 0,
	@COD_PRODUCTO int = 0,
	@PRODUCTO varchar(150) = '',
	@DOSIS varchar(150) = '',
	@ENTRATAMIENTO boolean = true,
	@MEDICO_PRESCRIPTOR varchar(150) = '',
	@CODIGO_VEEVA int = 0,
	@ESPECIALIDAD_MEDICO_PRESCRIPTOR int = 0,
)
as
BEGIN
	Update Productos Set
	COD_PRODUCTO=@COD_PRODUCTO,
	PRODUCTO=@PRODUCTO, 
	DOSIS=@DOSIS,
	ENTRATAMIENTO=@ENTRATAMIENTO,
	MEDICO_PRESCRIPTOR=@MEDICO_PRESCRIPTOR,
	CODIGO_VEEVA=@CODIGO_VEEVA,
	ESPECIALIDAD_MEDICO_PRESCRIPTOR=@ESPECIALIDAD_MEDICO_PRESCRIPTOR,
	WHERE Id=@Id
END

go
